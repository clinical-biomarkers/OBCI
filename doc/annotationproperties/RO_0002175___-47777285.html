    CONSTRUCT {
    in_taxon: a owl:ObjectProperty .
    ?witness rdfs:label ?label .

    ] .

    ?x present_in_taxon: ?taxon .

    "-",
    MD5(STR(?taxon))
    )) as ?witness)
    BIND(CONCAT(STR(?x), " in taxon ", STR(?taxon)) AS ?label)
    }"
